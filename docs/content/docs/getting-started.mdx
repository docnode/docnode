## 1. Define your nodes

```ts tab="Text Editor"
// nodes.ts
import { defineNode, number, string, boolean } from "docnode";

const TextNode = defineNode({
  type: "text",
  state: {
    text: string(""),
    bold: boolean(false),
    italic: boolean(false),
    underline: boolean(false),
  },
});

const ParagraphNode = defineNode({
  type: "paragraph",
  state: {
    indent: number(0),
  },
});

const HeadingNode = defineNode({
  type: "heading",
  state: {
    indent: number(0),
    level: number(1),
  },
});
```

```ts tab="File Explorer"
// nodes.ts
import { defineNode, string } from "docnode";
// We are going to see later how to define custom state definitions.
import { date } from "./your-custom-state-definitions";

const FileNode = defineNode({
  type: "file",
  state: {
    name: string(""),
    extension: string(""),
    updatedAt: date(),
  },
});

const FolderNode = defineNode({
  type: "folder",
  state: {
    name: string(""),
  },
});
```

```ts tab="Todo List"
// nodes.ts
import { defineNode, string, boolean } from "docnode";
// We are going to see later how to define custom state definitions.
import { date } from "./your-custom-state-definitions";

const TodoItemNode = defineNode({
  type: "todo-item",
  state: {
    title: string(""),
    description: string(""),
    completed: boolean(false),
    dueDate: date(),
  },
});

const TodoListNode = defineNode({
  type: "todo-list",
  state: {
    name: string(""),
    color: string("#f0f0f0"),
  },
});
```

## 2. Define your extensions

```ts tab="Text Editor"
// extensions.ts
import type { Extension } from "docnode";
import { ParagraphNode, HeadingNode } from "./nodes";

const RichTextBaseExtension: Extension = {
  nodes: [ParagraphNode, HeadingNode],
  // Let's suppose that you want to ensure that your
  // document always starts with a heading node.
  register: (doc) => {
    doc.onNormalize(() => {
      if (!doc.root.first?.is(HeadingNode)) {
        doc.root.prepend(doc.createNode(HeadingNode));
      }
    });
  },
};
```

## 3. Create your document

```ts tab="Text Editor"
import { Doc } from "docnode";
import { RichTextBaseExtension } from "./extensions";

const doc = new Doc({ extensions: [RichTextBaseExtension] });
```

```ts tab="Todo List"
import { Doc } from "docnode";
import { TodoListBaseExtension } from "./extensions";

const doc = new Doc({ extensions: [TodoListBaseExtension] });
```

```ts tab="File Explorer"
import { Doc } from "docnode";
import { FileExplorerBaseExtension } from "./extensions";

const doc = new Doc({ extensions: [FileExplorerBaseExtension] });
```

## 5. Edit your document

```ts
// A handler event example:

function setTodoItemDone(todoItemNode: DocNode<typeof TodoItemNode>) {
  node.state.completed.set(true);
}

function duplicateTodoItem(todoItemNode: DocNode<typeof TodoItemNode>) {
  todoItemNode.copy(todoItemNode, "after");
}
```
